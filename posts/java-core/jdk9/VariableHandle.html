public class LockFreeStack<T> {
  private static class Node<T> {
      final T item;
      Node<T> next;
      
      Node(T item) {
          this.item = item;
      }
  }
  
  private Node<T> head;
  private static final VarHandle HEAD_HANDLE;
  
  static {
      try {
          HEAD_HANDLE = MethodHandles.lookup()
              .findVarHandle(LockFreeStack.class, "head", Node.class);
      } catch (ReflectiveOperationException e) {
          throw new Error(e);
      }
  }
  
  public void push(T item) {
      Node<T> newNode = new Node<>(item);
      Node<T> currentHead;
      
      do {
          currentHead = (Node<T>) HEAD_HANDLE.getAcquire(this);
          newNode.next = currentHead;
      } while (!HEAD_HANDLE.compareAndSet(this, currentHead, newNode));
  }
  
  public T pop() {
      Node<T> currentHead;
      Node<T> newHead;
      
      do {
          currentHead = (Node<T>) HEAD_HANDLE.getAcquire(this);
          if (currentHead == null) {
              return null;
          }
          newHead = currentHead.next;
      } while (!HEAD_HANDLE.compareAndSet(this, currentHead, newHead));
      
      return currentHead.item;
  }
}

public class StateMachine {
  public enum State {
      IDLE, RUNNING, PAUSED, STOPPED
  }
  
  private State currentState = State.IDLE;
  private static final VarHandle STATE_HANDLE;
  
  static {
      try {
          STATE_HANDLE = MethodHandles.lookup()
              .findVarHandle(StateMachine.class, "currentState", State.class);
      } catch (ReflectiveOperationException e) {
          throw new Error(e);
      }
  }
  
  public boolean start() {
      return STATE_HANDLE.compareAndSet(this, State.IDLE, State.RUNNING);
  }
  
  public boolean pause() {
      return STATE_HANDLE.compareAndSet(this, State.RUNNING, State.PAUSED);
  }
  
  public boolean resume() {
      return STATE_HANDLE.compareAndSet(this, State.PAUSED, State.RUNNING);
  }
  
  public boolean stop() {
      State current = (State) STATE_HANDLE.getAcquire(this);
      return current != State.STOPPED && 
             STATE_HANDLE.compareAndSet(this, current, State.STOPPED);
  }
  
  public State getState() {
      return (State) STATE_HANDLE.getAcquire(this);
  }
}
